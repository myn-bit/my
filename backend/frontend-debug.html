Проблема: Фронтенд отправляет Bearer undefined

Шаг 1: Проверьте код сохранения токена

// ✅ ПРАВИЛЬНО:
const response = await fetch('/api/login', {
    method: 'POST',
    body: JSON.stringify({ email, password })
});
const data = await response.json();

// Проверяем, что токен есть
if (data.token) {
    // Сохраняем в localStorage
    localStorage.setItem('token', data.token);
    console.log('Токен сохранен:', data.token.substring(0, 30) + '...');
} else {
    console.error('Токен не получен от сервера');
}

// ❌ НЕПРАВИЛЬНО:
localStorage.setItem('token', response.token); // без проверки

Шаг 2: Проверьте код получения токена

// ✅ ПРАВИЛЬНО:
function getAuthToken() {
    const token = localStorage.getItem('token');
    
    // Проверяем, что токен не "undefined" или "null"
    if (!token || token === 'undefined' || token === 'null' || token === '') {
        console.error('Токен не найден или равен undefined/null');
        return null;
    }
    
    return token;
}

// ❌ НЕПРАВИЛЬНО:
const token = localStorage.getItem('token'); // без проверки

Шаг 3: Проверьте отправку запроса

// ✅ ПРАВИЛЬНО:
async function makeRequest(url, options = {}) {
    const token = getAuthToken();
    
    if (!token) {
        // Перенаправляем на страницу входа
        window.location.href = '/login';
        throw new Error('Токен не найден');
    }
    
    const headers = {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${token}`
    };
    
    const response = await fetch(url, {
        ...options,
        headers: { ...headers, ...options.headers }
    });
    
    return response;
}
